    rfc_auc, rsf_aucs = experiments.rsf_rfc(df_syn, censor_vals, time)
    rfc_auc_mean += rfc_auc
    rsf_auc_mean += np.asarray(rsf_aucs)
rfc_auc_mean /= num_experiments
